341static void taskMedirAceleracion()
342{
343 while (true)
344 {
345 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
346
347 AnalogInputReadSingle(CH0, &VOLTAJE_X); //lecturas en mV
348 AnalogInputReadSingle(CH1, &VOLTAJE_Y);
349 AnalogInputReadSingle(CH2, &VOLTAJE_Z);
350
351 VOLTAJE_X = VOLTAJE_X/1000; //PASO A VOLTS
352 VOLTAJE_Y = VOLTAJE_Y/1000;
353 VOLTAJE_Z = VOLTAJE_Z/1000;
354
355 AC_X = (VOLTAJE_X - 1.65)*(1/SENSIBILIDAD_AC);
356 AC_Y = (VOLTAJE_Y - 1.65)*(1/SENSIBILIDAD_AC);
357 AC_Z = (VOLTAJE_Z - 1.65)*(1/SENSIBILIDAD_AC);
358
359 AC_total = AC_X + AC_Y + AC_Z;
360 }
361}
