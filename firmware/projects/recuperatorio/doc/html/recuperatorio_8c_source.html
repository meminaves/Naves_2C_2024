154static void taskMedirVelocidad(void *pvParameter)
155{
156 while(true)
157 {
158 DISTANCIA_ANTERIOR = DISTANCIA;
159 DISTANCIA = HcSr04ReadDistanceInCentimeters();
160
161 if (DISTANCIA < 1000) //Si el vehículo se encuentra a menos de 1000 cm (10 m)...
162 {
163 VELOCIDAD = (DISTANCIA_ANTERIOR - DISTANCIA)/(0.1*100); //delta de desplazamiento sobre el tiempo que
164 controlarLeds(); //transcurrió entre ambos puntos (es decir el tiempo de muestreo)
165 //Además se divide por 100 para convertir de cm a m
166 if (VELOCIDAD > VELOCIDAD_MAX)
167 {
168 VELOCIDAD_MAX = VELOCIDAD;
169 }
170 }
171 else
172 {
173 LedsOffAll(); //si no hay vehículo se apagan los leds
174 VELOCIDAD_MAX = -1; //La velocidad no podria ser negativa porque asumo que el vehiculo no retrocede según la consigna
175 //Reinicio la v_max para el proximo camión
176 }
177
178 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
179 }

190static void pesarVehiculoTask(void*param)
191{
192 uint16_t contador = 0;
193
194 while (true)
195 {
196 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
197
198 uint16_t voltaje_galga_1;
199 uint16_t voltaje_galga_2;
200
201 if (VELOCIDAD == 0) //Si el vehículo está detenido...
202 {
203 if (contador < BUFFER_SIZE)
204 {
205 AnalogInputReadSingle(CH_GALGA1, &voltaje_galga_1); //salida en mV
206 AnalogInputReadSingle(CH_GALGA2, &voltaje_galga_2);
207
208 PESOS_GALGA_1[contador] = (voltaje_galga_1 * 20000)/(3.3*1000); //Paso a volts acá directamente dividendo por mil
209 PESOS_GALGA_2[contador] = (voltaje_galga_2 * 20000)/(3.3*1000);
210
211 contador++;
212 }
213 else
214 {
215 float Promedio1 = 0;
216 float Promedio2 = 0;
217
218 contador = 0;
219
220 for (int i = 0; i < BUFFER_SIZE; i++)
221 {
222 Promedio1 = PESOS_GALGA_1[i] + Promedio1;
223 Promedio2 = PESOS_GALGA_2[i] + Promedio2;
224 }
225
226 float PESO_PROMEDIO_1 = Promedio1 / BUFFER_SIZE;
227 float PESO_PROMEDIO_2 = Promedio2 / BUFFER_SIZE;
228
229 PESO = PESO_PROMEDIO_1 + PESO_PROMEDIO_2;
230
231 UartSendString(UART_PC, "Peso: ");
232 UartSendString(UART_PC, (char*)UartItoa(PESO,10));
233 UartSendString(UART_PC, "kg");
234 UartSendString(UART_PC, "\n");
235
236 UartSendString(UART_PC, "Velocidad Maxima: ");
237 UartSendString(UART_PC, (char*)UartItoa(VELOCIDAD_MAX,10));
238 UartSendString(UART_PC, "m/s");
239 UartSendString(UART_PC, "\n");
240 }
241 }
242 }

249void manejarBarrera()
250{
251 uint8_t tecla;
252 UartReadByte(UART_PC, &tecla);
253 switch (tecla)
254 {
255 case 'o':
256 GPIOOn(GPIO_BARRERA);
257 UartSendByte(UART_PC, (char*)&tecla);
258 break;
259
260 case 'c':
261 GPIOOff(GPIO_BARRERA);
262 UartSendByte(UART_PC, (char*)&tecla);
263 break;
264 }
