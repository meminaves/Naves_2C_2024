109static void manejarLeds()
110{
111 LedsOffAll();
112
113 if (MEDIDA < 10)
114 {
115 LedsOffAll();
116 }
117 else if(MEDIDA>=10 && MEDIDA <= 20)
118 {
119 printf("LED_1 ON\n");
120 LedOn(LED_1);
121 }
122 else if(MEDIDA>=20 && MEDIDA <= 30)
123 {
124 printf("LED_1 ON\n");
125 LedOn(LED_1);
126
127 printf("LED_2 ON\n");
128 LedOn(LED_2);
129 }
130 else
131 {
132 printf("LED_ON_ALL\n");
133 LedOn(LED_1);
134 LedOn(LED_2);
135 LedOn(LED_3);
136 }
137
138 vTaskDelay(CONFIG_BLINK_PERIOD_LED / portTICK_PERIOD_MS);
139}

150static void taskMostrarMedida(void *pvParameter)
151{
