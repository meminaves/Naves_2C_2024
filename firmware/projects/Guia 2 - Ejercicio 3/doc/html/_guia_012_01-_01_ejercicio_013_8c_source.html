223void detectarTeclas()
224{
225 uint8_t tecla;
226 UartReadByte(UART_PC, &tecla);
227 switch (tecla)
228 {
229 case 'O':
230 ON = !ON;
231 UartSendByte(UART_PC, (char*)&tecla);
232 break;
233
234 case 'H':
235 HOLD = !HOLD;
236 UartSendByte(UART_PC, (char*)&tecla);
237 break;
238 }
